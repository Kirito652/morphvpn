# MorphVPN v0.2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use compose:subagent (recommended) or compose:execute to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bring MorphVPN from experimental prototype to v0.2 by implementing rekey mechanism, TOML config file, cookie master key externalization, and integration tests.

**Architecture:** Four independent subsystems implemented sequentially: (1) Rekey — epoch key rotation via control frames when nonce space depletes, (2) Config — TOML-based configuration replacing hardcoded values, (3) Cookie key — externalize master key from hardcoded `[0xA5;32]`, (4) Integration tests — end-to-end handshake + data transfer verification.

**Tech Stack:** Rust, tokio, snow (Noise), chacha20poly1305, x25519-dalek, toml, serde, clap

pub fn handle_rekey_init(&mut self, epoch: u32, public_key: [u8; 32]) -> Result<Option<Bytes>> {
    let (new_tx, new_rx) = derive_epoch_keys(
        self._role,
        epoch.wrapping_add(1),
        &self._psk,
        self._handshake_hash.as_ref(),
    )?;
    self.tx_epoch = new_tx;
    self.rx_epoch = new_rx;
    self.data_tx_nonce = 0;
    self.control_tx_nonce = 0;
    self.data_replay = ReplayWindow2048::default();
    self.control_replay = ReplayWindow2048::default();

    let resp = self.send_control(
        ControlFrame::RekeyResp {
            epoch: self.tx_epoch.epoch,
            public_key: self.tx_epoch.mask_key,
        },
        8,
    )?;
    Ok(Some(resp))
}

pub fn handle_rekey_resp(&mut self, epoch: u32, public_key: [u8; 32]) -> Result<()> {
    let (new_tx, new_rx) = derive_epoch_keys(

    self.rx_epoch = new_rx;
    self.data_tx_nonce = 0;
    self.control_tx_nonce = 0;
    self.data_replay = ReplayWindow2048::default();

    let bytes = hex::decode(hex_str.trim())

fn full_handshake_and_data_transfer() {
    let client_id = StaticIdentity::generate();
    let server_id = StaticIdentity::generate();

    // Client creates init
    let client_tag = [0xAA; 12];
    let (pending_client, init_packet) = PendingClientHandshake::new(
        client_id.clone(),
        TEST_PSK,
        server_id.public,
        "10.8.0.5".parse().unwrap(),
        client_tag,
    )
    .unwrap();

    // Server processes init, creates response
    let server_tag = [0xBB; 12];
    let (pending_server, resp_packet) = PendingServerHandshake::from_init(
        &server_id,
        &TEST_PSK,
        "127.0.0.1:5001".parse().unwrap(),
        server_tag,
        &init_packet,
    )
    .unwrap();

    // Client processes response, creates finish
    let (mut client_session, finish_packet) = pending_client.into_established(resp_packet).unwrap();

    // Server processes finish
    let mut server_session = pending_server.complete(finish_packet, TEST_PSK).unwrap();

    // Assign IPs
    client_session.assign_ip("10.8.0.5".parse().unwrap());
    server_session.assign_ip("10.8.0.1".parse().unwrap());

    // Client sends data
    let test_payload = b"Hello from MorphVPN client!";
    let encrypted = client_session
        .send_data(bytes::Bytes::from_static(test_payload), 8)
        .unwrap();
    assert!(encrypted.len() > test_payload.len());

    // Server receives data
    let event = server_session.open_inbound(encrypted).unwrap();
    match event {
        morphvpn_protocol::session::SessionEvent::Data(payload) => {
            assert_eq!(payload.as_ref(), test_payload);
        }
        other => panic!("expected Data event, got {:?}", other),
    }

    // Server sends data back
    let reply_payload = b"Hello from MorphVPN server!";
    let encrypted_reply = server_session
        .send_data(bytes::Bytes::from_static(reply_payload), 8)
        .unwrap();

    // Client receives reply
    let event = client_session.open_inbound(encrypted_reply).unwrap();

            assert_eq!(payload.as_ref(), reply_payload);
        }
        other => panic!("expected Data event, got {:?}", other),
    }
}

#[test]
fn rekey_rotates_keys_and_continues_communication() {
    let mut pair = common::establish_pair();

    // Send some data before rekey
    let payload1 = b"before rekey";
    let encrypted1 = pair.client
        .send_data(bytes::Bytes::from_static(payload1), 8)
        .unwrap();
    let event = pair.server.open_inbound(encrypted1).unwrap();
    match event {
        morphvpn_protocol::session::SessionEvent::Data(data) => {
            assert_eq!(data.as_ref(), payload1);
        }
        other => panic!("expected Data, got {:?}", other),
    }

    // Force rekey by exhausting nonce
    pair.client.data_tx_nonce = u64::MAX - 100;
    let rekey_init = pair.client.advance_rekey().unwrap().unwrap();

    // Server processes rekey init
    let event = pair.server.open_inbound(rekey_init).unwrap();
    match event {
        morphvpn_protocol::session::SessionEvent::Control(frame) => {
            match frame {
                morphvpn_protocol::wire::ControlFrame::RekeyInit { epoch, .. } => {
                    let resp = pair.server.handle_rekey_init(epoch, [0; 32]).unwrap().unwrap();
                    // Client processes rekey response
                    let event2 = pair.client.open_inbound(resp).unwrap();
                    match event2 {
                        morphvpn_protocol::session::SessionEvent::Control(
                            morphvpn_protocol::wire::ControlFrame::RekeyResp { epoch, .. },
                        ) => {
                            pair.client.handle_rekey_resp(epoch, [0; 32]).unwrap();
                        }
                        other => panic!("expected RekeyResp, got {:?}", other),
                    }
                }
                other => panic!("expected RekeyInit, got {:?}", other),
            }
        }
        other => panic!("expected Control, got {:?}", other),
    }

    // Verify nonce counters reset
    assert_eq!(pair.client.data_tx_nonce, 0);
    assert_eq!(pair.server.data_tx_nonce, 0);

    // Send data after rekey
    let payload2 = b"after rekey";
    let encrypted2 = pair.client
        .send_data(bytes::Bytes::from_static(payload2), 8)
        .unwrap();
    let event = pair.server.open_inbound(encrypted2).unwrap();
    match event {
        morphvpn_protocol::session::SessionEvent::Data(data) => {
            assert_eq!(data.as_ref(), payload2);
        }
        other => panic!("expected Data after rekey, got {:?}", other),
    }
}
